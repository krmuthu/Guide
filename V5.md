# Part A — Agent Flow Architecture

## 1. Two Entry Paths

The agent accepts two types of input. Both converge into the same retrieval loop but with different seeds.

```
PATH A: Ticket-First (Jira available)
══════════════════════════════════════
Developer: "Implement BMT-4521"
  or: ome_rag_guide({ query: "...", ticket_id: "BMT-4521" })

  1. Fetch ticket via get_ticket_context MCP
  2. Run extractCodeIdentifier on ticket summary + description
     → PascalCase: OrderSubmissionService, RetryPolicy
     → UPPER_SNAKE: ENABLE_RETRY_FLAG
     → Action verbs: implement, retry
     → Synonyms: retry → retryable, retryHandler, retryPolicy
  3. Detect sparse ticket → enrich from parent/epic if needed
  4. Seed retrieval with:
     → find_symbol(each PascalCase identifier)
     → find_symbol(each UPPER_SNAKE identifier)
     → search_exact_match(each synonym variant)
     → search_semantic_code(action verbs as query)
     → get_ticket_context for linked tickets

PATH B: Query-First (no ticket)
════════════════════════════════
Developer: "How does bond trading auth work?"
  or: "Add pagination to bond search"
  or: "NPE in SettlementService line 42"

  1. Extract identifiers from query text (same extractCodeIdentifier)
  2. Seed retrieval with:
     → search_semantic_code(full query)
     → find_symbol(each extracted identifier)
     → search_exact_match(each synonym variant)

BOTH PATHS CONVERGE:
  → Agentic retrieval loop
  → Generation pipeline (per mode)
  → Response
```

```
┌─────────────────────────────────────────────────────────────────────┐

│                                                                     │
│  Tier 1 (1.5B, ~50ms):                                             │
│    • Classify intent → understand / implement / diagnose            │

├─────────────────────────────────────────────────────────────────────┤
│  STEP 2: SEED RETRIEVAL                                             │
│                                                                     │

│                                                                     │

│    search_confluence, find_symbol, get_code_context,                │

│    get_repo_config, lookup_glossary, read_full_file (config only)   │
│                                                                     │

│  STEP 4: POST-RETRIEVAL                                            │

│  Tier 2 (~2-3s):                                                    │
│    • Convention detection (from reference file chunks)               │
│    • Architecture pattern identification                            │
│    • Cross-repo context assembly                                    │

│    • Route complexity → decide Tier 2 or Tier 3 for generation      │
```

## 3. Confidence System

The loop doesn’t exit on a fixed iteration count. It exits when **confidence** reaches the threshold for the mode.

```typescript
// src/guide/retrieval/confidence.ts

export interface ConfidenceScore {
  score: number;          // 0.0 - 1.0
  level: 'low' | 'medium' | 'high';
  reasons: string[];      // Why this score
  gaps: string[];          // What's still missing
}

export function computeConfidence(
  mode: GuideMode,
  store: DataStore
): ConfidenceScore {
  const checks: Array<{ name: string; met: boolean; weight: number }> = [];

  // ── Universal checks ──
  checks.push({
    name: 'has_search_results',
    met: store.searchResults.length > 0,
    weight: 0.15,
  });
  checks.push({
    name: 'has_primary_symbol',
    met: store.resolvedSymbols.size > 0,
    weight: 0.10,
  });
  checks.push({
    name: 'has_project_info',
    met: store.projectInfo.size > 0,
    weight: 0.05,
  });

  // ── Mode-specific checks ──
  switch (mode) {
    case 'implement':
      checks.push({
        name: 'has_target_files',
        met: store.getFilesByRole('target').length > 0,
        weight: 0.20,
      });
      checks.push({
        name: 'has_type_definitions',
        met: store.typeDefinitions.size > 0 || store.unresolvedTypes.size === 0,
        weight: 0.10,
      });
      checks.push({
        name: 'has_caller_signatures',
        met: store.callerSignatures.length > 0,
        weight: 0.10,
      });
      checks.push({
        name: 'has_convention_reference',
        met: store.getFilesByRole('reference').length > 0,
        weight: 0.10,
      });
      checks.push({
        name: 'has_ticket_or_query_context',
        met: store.ticketContext !== null || store.query.length > 20,
        weight: 0.10,
      });
      checks.push({
        name: 'no_unresolved_cross_repo',
        met: store.unresolvedCrossRepoRefs.length === 0,
        weight: 0.05,
      });
      checks.push({
        name: 'has_architecture_pattern',
        met: store.architecturePatterns.length > 0,
        weight: 0.05,
      });
      break;

    case 'understand':
      checks.push({
        name: 'has_assembled_context',
        met: store.assembledFiles.size > 0,
        weight: 0.25,
      });
      checks.push({
        name: 'has_call_chain',
        met: store.callerSignatures.length > 0 || store.calleeSignatures.length > 0,
        weight: 0.15,
      });
      checks.push({
        name: 'has_architecture_context',
        met: store.architecturePatterns.length > 0,
        weight: 0.10,
      });
      checks.push({
        name: 'cross_repo_explored',
        met: store.crossRepoRefs.length === 0 || store.crossRepoExplored,
        weight: 0.10,
      });
      break;

    case 'diagnose':
      checks.push({
        name: 'has_error_location',
        met: store.getFilesByRole('target').length > 0,
        weight: 0.25,
      });
      checks.push({
        name: 'has_call_chain_to_error',
        met: store.callerSignatures.length > 0,
        weight: 0.15,
      });
      checks.push({
        name: 'has_recent_changes',
        met: store.recentChanges.length > 0,
        weight: 0.10,
      });
      checks.push({
        name: 'has_config_context',
        met: store.repoConfig !== null || !store.signals.some(s => s.type === 'config_dependency'),
        weight: 0.05,
      });
      checks.push({
        name: 'cross_repo_explored',
        met: store.crossRepoRefs.length === 0 || store.crossRepoExplored,
        weight: 0.10,
      });
      break;
  }

  const totalWeight = checks.reduce((sum, c) => sum + c.weight, 0);
  const achievedWeight = checks.filter(c => c.met).reduce((sum, c) => sum + c.weight, 0);
  const score = achievedWeight / totalWeight;

  const gaps = checks.filter(c => !c.met).map(c => c.name);
  const reasons = checks.filter(c => c.met).map(c => c.name);

  return {
    score,
    level: score >= 0.85 ? 'high' : score >= 0.60 ? 'medium' : 'low',
    reasons,
    gaps,
  };
}

// Mode-specific thresholds
export const CONFIDENCE_THRESHOLDS: Record<GuideMode, number> = {
  implement: 0.80,    // Need high confidence — generating code
  understand: 0.65,   // Can answer with partial context
  diagnose: 0.70,     // Need error location + context
};
```

## 4. Data Store

The DataStore accumulates everything retrieved across iterations. The LLM only receives what it NEEDS for the current task — not the entire store.

### Why a Store

```
PROBLEM WITHOUT STORE:
  Iteration 1: search returns 10 results (2,000 tokens)
  Iteration 2: assemble 3 files (3,000 tokens) + search returns 8 more (1,500 tokens)
  Iteration 3: SCIP callers (500 tokens) + type defs (800 tokens)

  Generation prompt: ALL of the above = 7,800 tokens
  But the LLM only needs: 3 assembled files + relevant types = 4,300 tokens

PROBLEM WITH NAIVE ACCUMULATION:
  Every iteration, reasoning model sees growing context
  By iteration 5, the reasoning prompt itself is 8,000+ tokens

SOLUTION — DATA STORE:
  Store everything indexed by category.
  Reasoning model sees SUMMARIES (one-line per item).
  Generation model receives SELECTED data (full content for relevant items only).
  Response includes REFERENCES to stored data (file paths, evidence).
```

```typescript
// src/guide/store/data_store.ts

export class DataStore {
  readonly sessionId: string;

  // ── Indexed Collections ──

  /** Search results — deduplicated by filePath + chunkId */
  readonly searchResults: IndexedCollection<SearchResult>;

  /** Assembled file contexts — by filePath */
  readonly assembledFiles: Map<string, AssembledFileContext>;

  /** File skeletons — by filePath (lighter than assembled) */
  readonly skeletons: Map<string, FileSkeleton>;

  /** Resolved symbols — by symbol name */
  readonly resolvedSymbols: Map<string, ResolvedSymbol>;

  /** Type definitions — by type name */
  readonly typeDefinitions: Map<string, TypeDefinition>;

  /** Unresolved types — referenced but not yet fetched */
  readonly unresolvedTypes: Set<string>;

  /** Caller signatures — external methods that call into our targets */
  readonly callerSignatures: ExternalSignature[];

  /** Callee signatures — methods our targets call */
  readonly calleeSignatures: ExternalSignature[];

  /** Interface contracts — what our classes must satisfy */
  readonly interfaceContracts: InterfaceContract[];

  /** Cross-repo references — detected but possibly not yet explored */
  readonly crossRepoRefs: CrossRepoRef[];
  readonly unresolvedCrossRepoRefs: CrossRepoRef[];
  crossRepoExplored: boolean = false;

  /** Architecture patterns found */
  readonly architecturePatterns: ArchitecturePattern[];

  /** Ticket context */
  ticketContext: TicketContext | null = null;
  ticketIdentifiers: ExtractedIdentifiers | null = null;

  /** Similar PRs */
  readonly similarPRs: PRResult[];

  /** Conventions detected */
  conventions: string | null = null;

  /** Project info — per repo */
  readonly projectInfo: Map<string, ProjectInfo>;

  /** Repo config */
  repoConfig: any = null;

  /** Recent changes */
  readonly recentChanges: RecentChange[];

  /** Signals from tool responses */
  readonly signals: RetrievalSignal[];

  /** Glossary expansions */
  readonly glossary: Map<string, string>;

  // ── Generated Outputs (populated during generation) ──

  /** Blueprint from planning phase */
  blueprint: ImplementationBlueprint | null = null;

  /** Generated code changes — by filePath */
  readonly codeChanges: Map<string, CodeChange[]>;

  /** Generated test changes — by test filePath */
  readonly testChanges: Map<string, TestChange[]>;

  /** Verification results */
  verificationResult: VerificationResult | null = null;

  // ── Metadata ──

  readonly query: string;
  readonly mode: GuideMode;
  readonly actionLog: ActionLogEntry[];
  totalToolCalls: number = 0;
  totalLLMCalls: { tier1: number; tier2: number; tier3: number } = { tier1: 0, tier2: 0, tier3: 0 };
  totalTokensAccumulated: number = 0;

  constructor(query: string, mode: GuideMode) {
    this.sessionId = crypto.randomUUID();
    this.query = query;
    this.mode = mode;
    this.searchResults = new IndexedCollection('chunkId');
    this.assembledFiles = new Map();
    this.skeletons = new Map();
    this.resolvedSymbols = new Map();
    this.typeDefinitions = new Map();
    this.unresolvedTypes = new Set();
    this.callerSignatures = [];
    this.calleeSignatures = [];
    this.interfaceContracts = [];
    this.crossRepoRefs = [];
    this.unresolvedCrossRepoRefs = [];
    this.architecturePatterns = [];
    this.similarPRs = [];
    this.projectInfo = new Map();
    this.recentChanges = [];
    this.signals = [];
    this.glossary = new Map();
    this.codeChanges = new Map();
    this.testChanges = new Map();
    this.actionLog = [];
  }

  // ── Absorb MCP tool response ──

  absorbSearchResults(results: SearchResult[]): void {
    for (const r of results) {
      this.searchResults.add(r);
      // Track unresolved types from search content
      const types = extractTypeReferences(r.content ?? '');
      for (const t of types) {
        if (!this.typeDefinitions.has(t)) this.unresolvedTypes.add(t);
      }
    }
    this.totalTokensAccumulated += estimateTokens(JSON.stringify(results));
  }

  absorbAssembledFile(ctx: AssembledFileContext): void {
    this.assembledFiles.set(ctx.filePath, ctx);
    // Absorb SCIP data
    this.callerSignatures.push(...ctx.scipEnrichment.callerSignatures);
    this.calleeSignatures.push(...ctx.scipEnrichment.calleeSignatures);
    this.interfaceContracts.push(...ctx.scipEnrichment.implementedInterfaces);
    for (const type of ctx.scipEnrichment.referencedTypes) {
      this.typeDefinitions.set(type.typeName, type);
      this.unresolvedTypes.delete(type.typeName);
    }
    for (const ref of ctx.scipEnrichment.crossRepoRefs) {
      this.crossRepoRefs.push({
        repoSlug: ref.repoSlug,
        reason: ref.relationship,
        filesFound: [ref.filePath],
      });
    }
    this.totalTokensAccumulated += ctx.tokenEstimate;
  }

  absorbSymbol(symbol: ResolvedSymbol): void {
    this.resolvedSymbols.set(symbol.name, symbol);
  }

  absorbTypeDefinition(typeDef: TypeDefinition): void {
    this.typeDefinitions.set(typeDef.typeName, typeDef);
    this.unresolvedTypes.delete(typeDef.typeName);
  }

  absorbArchitecturePattern(pattern: ArchitecturePattern): void {
    this.architecturePatterns.push(pattern);
  }

  absorbSignal(signal: RetrievalSignal): void {
    this.signals.push(signal);
  }

  // ── Query helpers ──

  getFilesByRole(role: string): AssembledFileContext[] {
    return [...this.assembledFiles.values()].filter(f => f.role === role);
  }

  getSkeletonsSummary(): string {
    const files = [...this.assembledFiles.values(), ...this.skeletons.entries()];
    // One-line per file for reasoning prompt
    return [...this.assembledFiles.values()].map(f =>
      `${f.filePath} (${f.role}, ${f.matchedChunks.length} chunks, ${f.skeleton.methodSignatures.length} methods)`
    ).join('\n');
  }

  // ── Compact summary for reasoning model ──

  toReasoningSummary(): string {
    // Compact — one line per item, just enough for routing decisions
    const sections: string[] = [];

    sections.push(`Files: ${this.assembledFiles.size} assembled, ${this.skeletons.size} skeletons`);
    if (this.assembledFiles.size > 0) {
      sections.push(this.getSkeletonsSummary());
    }

    sections.push(`Symbols: ${this.resolvedSymbols.size} resolved`);
    if (this.unresolvedTypes.size > 0) {
      sections.push(`Unresolved types: ${[...this.unresolvedTypes].join(', ')}`);
    }

    if (this.callerSignatures.length > 0) {
      sections.push(`Callers: ${this.callerSignatures.slice(0, 5).map(c => c.symbol).join(', ')}`);
    }
    if (this.calleeSignatures.length > 0) {
      sections.push(`Callees: ${this.calleeSignatures.slice(0, 5).map(c => c.symbol).join(', ')}`);
    }

    if (this.crossRepoRefs.length > 0) {
      sections.push(`Cross-repo: ${this.crossRepoRefs.map(r => r.repoSlug).join(', ')}`);
    }
    if (this.architecturePatterns.length > 0) {
      sections.push(`Patterns: ${this.architecturePatterns.map(p => p.name).join(', ')}`);
    }
    if (this.ticketContext) {
      sections.push(`Ticket: ${this.ticketContext.ticketId} — ${this.ticketContext.summary}`);
    }
    if (this.similarPRs.length > 0) {
      sections.push(`Similar PRs: ${this.similarPRs.map(p => `PR-${p.pr_number}`).join(', ')}`);
    }

    sections.push(`Signals: ${this.signals.length} active`);
    for (const s of this.signals.slice(-5)) {
      sections.push(`  [${s.type}] ${formatSignalCompact(s)}`);
    }

    sections.push(`Tokens accumulated: ~${this.totalTokensAccumulated}`);

    return sections.join('\n');
  }

  // ── Select data for generation prompt (only what's needed) ──

  selectForGeneration(filePath: string): GenerationContext {
    const assembled = this.assembledFiles.get(filePath);
    if (!assembled) throw new Error(`File ${filePath} not assembled`);

    return {
      targetFile: assembled,
      // Only callers/callees relevant to THIS file
      relevantCallers: this.callerSignatures.filter(c =>
        c.filePath !== filePath && assembled.matchedChunks.some(chunk =>
          this.isCallerOf(c, chunk)
        )
      ),
      relevantCallees: this.calleeSignatures.filter(c =>
        c.filePath !== filePath
      ),
      // Only types used in THIS file
      relevantTypes: [...this.typeDefinitions.values()].filter(t =>
        assembled.matchedChunks.some(chunk =>
          chunk.content.includes(t.typeName)
        )
      ),
      // Interface contracts for THIS file
      relevantContracts: this.interfaceContracts.filter(i =>
        i.filePath === filePath || assembled.skeleton.classDeclaration?.includes(i.interfaceName)
      ),
      conventions: this.conventions,
      ticket: this.ticketContext,
    };
  }

  // ── Persist to DB (for long sessions or crash recovery) ──

  async persistToDb(pool: Pool): Promise<void> {
    await pool.query(`
      INSERT INTO ome_agent.session_store (session_id, mode, query, store_data, updated_at)
      VALUES ($1, $2, $3, $4, NOW())
      ON CONFLICT (session_id) DO UPDATE SET store_data = $4, updated_at = NOW()
    `, [this.sessionId, this.mode, this.query, JSON.stringify(this.serialize())]);
  }

  static async loadFromDb(pool: Pool, sessionId: string): Promise<DataStore | null> {
    const result = await pool.query(
      `SELECT * FROM ome_agent.session_store WHERE session_id = $1`, [sessionId]
    );
    if (result.rows.length === 0) return null;
    return DataStore.deserialize(result.rows[0].store_data);
  }

  private serialize(): any { /* ... */ }
  private static deserialize(data: any): DataStore { /* ... */ }
}
```

# Part B — Retrieval Loop (Detail)

## 5. Entry Path Resolution

```typescript
// src/guide/retrieval/entry_resolver.ts

export interface ResolvedEntry {
  path: 'ticket' | 'query';
  identifiers: ExtractedIdentifiers;
  ticketContext: TicketContext | null;
  enrichedTicket: boolean;
  seedActions: RetrievalAction[];
}

export async function resolveEntryPath(
  params: GuideParams,
  ctx: AgentContext
): Promise<ResolvedEntry> {

  let ticketContext: TicketContext | null = null;
  let enrichedTicket = false;

  // ── PATH A: Ticket-first ──
  if (params.ticket_id) {
    ticketContext = await ctx.rag.getTicketContext(params.ticket_id);

    // Detect sparse ticket
    const isSparse = await ctx.tier1.classify(
      `Is this ticket sparse (less than 2 sentences of real technical detail)?
       Summary: "${ticketContext.summary}"
       Description: "${ticketContext.specificTask ?? ''}"
       AC: "${ticketContext.acceptanceCriteria ?? ''}"
       Answer: sparse or adequate`,
      { maxTokens: 10 }
    );

    if (isSparse.includes('sparse')) {
      ticketContext = await enrichSparseTicket(ticketContext, ctx);
      enrichedTicket = true;
    }
  }

  // ── Extract code identifiers from ticket OR query ──
  const sourceText = ticketContext
    ? `${ticketContext.summary} ${ticketContext.specificTask ?? ''} ${ticketContext.acceptanceCriteria ?? ''}`
    : params.query;

  const identifiers = extractCodeIdentifier(sourceText);

  // ── Build seed actions ──
  const seedActions: RetrievalAction[] = [];

  // Semantic search on full query (always)
  seedActions.push({ action: 'semantic_search', query: params.query });

  // If ticket has different wording, also search with ticket summary
  if (ticketContext && ticketContext.summary !== params.query) {
    seedActions.push({ action: 'semantic_search', query: ticketContext.summary });
  }

  // Find symbols from extracted identifiers
  for (const id of identifiers.pascalCase.slice(0, 3)) {
    seedActions.push({ action: 'find_symbol', symbol: id });
  }
  for (const id of identifiers.upperSnake.slice(0, 2)) {
    seedActions.push({ action: 'find_symbol', symbol: id });
  }

  // Exact search with synonyms
  for (const syn of identifiers.synonyms.slice(0, 3)) {
    seedActions.push({ action: 'search_exact_match', query: syn });
  }

  // Architecture search with action verbs
  if (identifiers.actionVerbs.length > 0) {
    const archQuery = identifiers.actionVerbs.slice(0, 2).join(' ') +
      ' ' + identifiers.pascalCase.slice(0, 1).join(' ');
    seedActions.push({ action: 'search_architecture', query: archQuery.trim() });
  }

  // Project info
  if (params.current_repo) {
    seedActions.push({ action: 'get_project_info', repo: params.current_repo });
  }

  // Current file
  if (params.current_file && params.current_repo) {
    seedActions.push({
      action: 'assemble_file_context',
      filePath: params.current_file,
      repoSlug: params.current_repo,
      role: 'target',
    });
  }

  // Glossary
  if (identifiers.domainTerms.length > 0) {
    seedActions.push({ action: 'lookup_glossary', terms: identifiers.domainTerms });
  }

  return {
    path: params.ticket_id ? 'ticket' : 'query',
    identifiers,
    ticketContext,
    enrichedTicket,
    seedActions,
  };
}
```

```typescript
// src/guide/retrieval/code_identifier_extractor.ts

export interface ExtractedIdentifiers {
  /** PascalCase: OrderService, BondDTO, RetryPolicy */
  pascalCase: string[];
  /** UPPER_SNAKE_CASE: ENABLE_RETRY_FLAG, MAX_RETRIES */
  upperSnake: string[];
  /** camelCase: submitOrder, getBondById */
  camelCase: string[];
  /** Action verbs: implement, add, refactor, fix, migrate */
  actionVerbs: string[];
  /** Synonym variants: retry → retryable, retryHandler, retryPolicy, retryConfig */
  synonyms: string[];
  /** Domain terms for glossary lookup: BMT, OMS, FX */
  domainTerms: string[];
}

export function extractCodeIdentifier(text: string): ExtractedIdentifiers {
  const NOISE_WORDS = new Set([
    'The', 'This', 'That', 'When', 'Where', 'How', 'What', 'Which',
    'Please', 'Should', 'Could', 'Would', 'Need', 'Want', 'Like',
    'TODO', 'FIXME', 'NOTE', 'DONE', 'WIP', 'ASAP', 'NULL', 'TRUE', 'FALSE',
    'POST', 'GET', 'PUT', 'DELETE', 'PATCH',
  ]);

  const MIN_LENGTH = 3;

  // PascalCase: starts with uppercase, has lowercase
  const pascalCase = [...new Set(
    (text.match(/\b[A-Z][a-z]+(?:[A-Z][a-z]+)+\b/g) ?? [])
      .filter(w => w.length >= MIN_LENGTH && !NOISE_WORDS.has(w))
  )];

  // UPPER_SNAKE_CASE: all caps with underscores
  const upperSnake = [...new Set(
    (text.match(/\b[A-Z][A-Z0-9]+(?:_[A-Z0-9]+)+\b/g) ?? [])
      .filter(w => w.length >= MIN_LENGTH && !NOISE_WORDS.has(w))
  )];

  // camelCase: starts with lowercase, has uppercase
  const camelCase = [...new Set(
    (text.match(/\b[a-z]+[A-Z][a-zA-Z0-9]*\b/g) ?? [])
      .filter(w => w.length >= MIN_LENGTH)
  )];

  // Action verbs
  const ACTION_VERBS = new Set([
    'add', 'create', 'implement', 'build', 'write', 'update', 'change',
    'modify', 'refactor', 'remove', 'delete', 'migrate', 'introduce',
    'fix', 'resolve', 'debug', 'rename', 'extract', 'split', 'merge',
    'move', 'replace', 'upgrade', 'deprecate', 'enable', 'disable',
    'configure', 'integrate', 'optimize', 'retry', 'paginate',
  ]);

  const words = text.toLowerCase().split(/\s+/);
  const actionVerbs = [...new Set(words.filter(w => ACTION_VERBS.has(w)))];

  // Synonym generation
  const synonyms: string[] = [];
  const SYNONYM_MAP: Record<string, string[]> = {
    'retry': ['retryable', 'retryHandler', 'retryPolicy', 'retryConfig', 'RetryTemplate'],
    'paginate': ['pagination', 'Pageable', 'PageRequest', 'PagedResult'],
    'auth': ['authentication', 'authorization', 'AuthService', 'AuthFilter', 'AuthInterceptor'],
    'cache': ['caching', 'CacheManager', 'CacheConfig', 'Cacheable'],
    'validate': ['validation', 'Validator', 'ValidationResult', 'Valid'],
    'circuit': ['CircuitBreaker', 'CircuitBreakerConfig', 'Resilience4j'],
    'kafka': ['KafkaListener', 'KafkaTemplate', 'KafkaConsumer', 'KafkaProducer'],
    'schedule': ['Scheduled', 'ScheduledTask', 'Scheduler', 'CronJob'],
  };

  for (const verb of actionVerbs) {
    if (SYNONYM_MAP[verb]) {
      synonyms.push(...SYNONYM_MAP[verb]);
    }
  }
  // Also generate variants from PascalCase identifiers
  for (const pc of pascalCase) {
    synonyms.push(pc + 'Service', pc + 'Controller', pc + 'Repository', pc + 'Config');
  }

  // Domain terms (3-4 char uppercase that might be acronyms)
  const domainTerms = [...new Set(
    (text.match(/\b[A-Z]{2,5}\b/g) ?? [])
      .filter(w => !NOISE_WORDS.has(w) && w.length >= 2)
  )];

  return { pascalCase, upperSnake, camelCase, actionVerbs, synonyms, domainTerms };
}
```

## 8. Tool Response Handling — Token-Aware Iteration

When an MCP tool returns a large response (e.g., `assemble_file_context` for a 500-line file with 8 matched chunks), we don’t dump it all into the next reasoning prompt.

### The Problem

```
assemble_file_context returns:
  8 matched chunks × ~300 tokens = 2,400 tokens
  12 method signatures × ~20 tokens = 240 tokens
  6 caller signatures = 180 tokens
  4 type definitions × ~100 tokens = 400 tokens
  Total: ~3,220 tokens

If we add this to the reasoning prompt, it bloats to 8,000+ tokens.
But the reasoning model only needs: "file assembled, 8 methods matched, 4 types resolved"
```

### Solution: Store Full Data, Summarize for Reasoning

```typescript
// src/guide/retrieval/tool_absorber.ts

export function absorbToolResult(
  store: DataStore,
  action: RetrievalAction,
  result: ToolResult
): void {
  // 1. Store FULL data in DataStore (for generation phase)
  switch (action.action) {
    case 'semantic_search':
    case 'exact_search':
      store.absorbSearchResults(result.data);
      break;
    case 'assemble_file_context':
      store.absorbAssembledFile(result.data);
      break;
    case 'get_file_skeleton':
      store.skeletons.set(result.data.filePath, result.data);
      break;
    case 'find_symbol':
      for (const sym of result.data) store.absorbSymbol(sym);
      break;
    case 'get_scip_type_definition':
      store.absorbTypeDefinition(result.data);
      break;
    case 'get_scip_callers':
      store.callerSignatures.push(...result.data);
      break;
    case 'get_scip_callees':
      store.calleeSignatures.push(...result.data);
      break;
    case 'get_scip_implementations':
      store.interfaceContracts.push(...result.data);
      break;
    case 'search_architecture':
      for (const pattern of result.data) store.absorbArchitecturePattern(pattern);
      break;
    case 'search_similar_prs':
      store.similarPRs.push(...result.data);
      break;
    case 'get_recent_changes':
      store.recentChanges.push(...result.data);
      break;
    case 'get_project_info':
      store.projectInfo.set(result.data.repoSlug, result.data);
      break;
    case 'get_repo_config':
      store.repoConfig = result.data;
      break;
    case 'lookup_glossary':
      for (const [term, meaning] of Object.entries(result.data)) {
        store.glossary.set(term as string, meaning as string);
      }
      break;
    // ... other tools
  }

  // 2. Extract signals (programmatic, no LLM)
  const signals = extractSignals(action, result);
  for (const signal of signals) store.absorbSignal(signal);

  // 3. Track metadata
  store.totalToolCalls++;
  store.actionLog.push({
    iteration: store.actionLog.length,
    action: action.action,
    success: result.success,
    summary: result.summary,
    tokensAdded: result.tokenEstimate,
  });
}
```

### Token-Aware Batched Processing for Large Responses

When the accumulated data exceeds what can be sent to the generation model in one call, we split intelligently.

```typescript
// src/guide/pipeline/token_manager.ts

export interface TokenBudget {
  contextWindow: number;          // Model's context window
  systemPromptReserve: number;    // ~1,000 tokens
  generationReserve: number;      // Output tokens reserved
  availableForInput: number;      // What's left for file context
}

export function computeTokenBudget(tier: ModelTier, mode: GuideMode): TokenBudget {
  const contextWindow = tier === 'tier2' ? 65_536 : 65_536; // Both Coder-Next and 122B have 64K
  const systemPromptReserve = 1_200;
  const generationReserve = mode === 'implement' ? 8_000 : mode === 'diagnose' ? 6_000 : 4_000;

  return {
    contextWindow,
    systemPromptReserve,
    generationReserve,
    availableForInput: contextWindow - systemPromptReserve - generationReserve,
  };
}

/**
 * Select data from the DataStore that fits within the token budget.
 * Priority order determines what gets included first.
 */
export function selectDataForPrompt(
  store: DataStore,
  filePath: string,
  budget: TokenBudget
): SelectedData {
  let remaining = budget.availableForInput;
  const selected: SelectedData = {
    targetChunks: [],
    skeleton: null,
    callerSignatures: [],
    calleeSignatures: [],
    typeDefinitions: [],
    conventions: null,
    ticket: null,
    blueprint: null,
  };

  const assembled = store.assembledFiles.get(filePath);
  if (!assembled) return selected;

  // ── P0: Matched chunks (must have — this is the code being modified) ──
  for (const chunk of assembled.matchedChunks) {
    const tokens = estimateTokens(chunk.content);
    if (remaining >= tokens) {
      selected.targetChunks.push(chunk);
      remaining -= tokens;
    }
  }

  // ── P0: File skeleton (imports + class + fields + method sigs) ──
  const skelTokens = estimateTokens(formatSkeleton(assembled.skeleton));
  if (remaining >= skelTokens) {
    selected.skeleton = assembled.skeleton;
    remaining -= skelTokens;
  }

  // ── P1: Blueprint instruction for this file ──
  if (store.blueprint) {
    const plan = store.blueprint.fileChangePlans.find(p => p.filePath === filePath);
    if (plan) {
      const planTokens = estimateTokens(JSON.stringify(plan));
      if (remaining >= planTokens) {
        selected.blueprint = plan;
        remaining -= planTokens;
      }
    }
  }

  // ── P1: Ticket ──
  if (store.ticketContext) {
    const ticketTokens = estimateTokens(formatTicket(store.ticketContext));
    if (remaining >= ticketTokens) {
      selected.ticket = store.ticketContext;
      remaining -= ticketTokens;
    }
  }

  // ── P1: Conventions ──
  if (store.conventions) {
    const convTokens = estimateTokens(store.conventions);
    if (remaining >= convTokens) {
      selected.conventions = store.conventions;
      remaining -= convTokens;
    }
  }

  // ── P2: Caller signatures (cross-file compatibility) ──
  for (const caller of assembled.scipEnrichment.callerSignatures) {
    const tokens = estimateTokens(caller.signature);
    if (remaining >= tokens) {
      selected.callerSignatures.push(caller);
      remaining -= tokens;
    }
  }

  // ── P2: Callee signatures ──
  for (const callee of assembled.scipEnrichment.calleeSignatures) {
    const tokens = estimateTokens(callee.signature);
    if (remaining >= tokens) {
      selected.calleeSignatures.push(callee);
      remaining -= tokens;
    }
  }

  // ── P3: Type definitions ──
  for (const type of assembled.scipEnrichment.referencedTypes) {
    const tokens = estimateTokens(type.definition);
    if (remaining >= tokens) {
      selected.typeDefinitions.push(type);
      remaining -= tokens;
    }
  }

  return selected;
}
```

## 19. Token-Aware MCP Response Processing

```typescript
// src/guide/retrieval/token_manager.ts

/**
 * When an MCP tool returns a large response, don't send it all
 * to the reasoning model. Store the full data, summarize for reasoning.
 */
export function processLargeResponse(
  response: any,
  maxReasoningTokens: number = 200
): { fullData: any; summary: string } {
  const fullTokens = estimateTokens(JSON.stringify(response));

  if (fullTokens <= maxReasoningTokens) {
    // Small enough — use as-is
    return { fullData: response, summary: JSON.stringify(response) };
  }

  // Large response — store full, generate compact summary
  if (Array.isArray(response)) {
    return {
      fullData: response,
      summary: `${response.length} results: ${response.slice(0, 3).map(r =>
        typeof r === 'object' ? (r.filePath ?? r.name ?? r.symbol ?? 'item') : String(r)
      ).join(', ')}${response.length > 3 ? ` +${response.length - 3} more` : ''}`,
    };
  }

  if (response.content && typeof response.content === 'string') {
    return {
      fullData: response,
      summary: `File: ${response.filePath ?? 'unknown'} (${response.content.split('\n').length} lines)`,
    };
  }

  return {
    fullData: response,
    summary: `Data: ${Object.keys(response).join(', ')} (~${fullTokens} tokens)`,
  };
}
```

## 20. Multi-Batch LLM Iteration for Large Data

```typescript
// src/guide/pipeline/batch_processor.ts

/**
 * If a single file's assembled context exceeds the generation budget,
 * split the chunks into batches and generate changes per batch.
 *
 * This is rare — only for very large files with many matched chunks.
 */
export async function generateWithBatching(
  filePath: string,
  store: DataStore,
  budget: TokenBudget,
  ctx: AgentContext
): Promise<CodeChange[]> {
  const assembled = store.assembledFiles.get(filePath)!;
  const totalTokens = assembled.tokenEstimate;

  if (totalTokens <= budget.availableForInput) {
    // Fits in one call — normal generation
    return generateSinglePass(filePath, store, ctx);
  }

  // Split matched chunks into batches that fit
  const batches = splitChunksIntoBatches(
    assembled.matchedChunks,
    budget.availableForInput * 0.6 // 60% for chunks, 40% for skeleton/SCIP/conventions
  );

  console.log(
    `[BatchProcessor] ${filePath}: ${assembled.matchedChunks.length} chunks split into ${batches.length} batches`
  );

  const allChanges: CodeChange[] = [];
  const carryForward: string[] = [];

  for (const batch of batches) {
    // Build prompt with this batch of chunks + skeleton + SCIP
    const batchAssembled: AssembledFileContext = {
      ...assembled,
      matchedChunks: batch,
    };

    const selected = selectDataForPromptWithCarryForward(
      batchAssembled, store, budget, carryForward
    );

    const response = await ctx.tier3.chat(
      buildFileGenerationPrompt(selected, filePath),
      { maxTokens: 8_000 }
    );

    const changes = parseCodeChanges(response.content);
    allChanges.push(...changes);

    // Carry forward: what was generated in this batch
    carryForward.push(
      ...changes.map(c => `${c.id}: ${c.action} at L${c.currentCode?.startLine ?? '?'} — ${c.description}`)
    );
  }

  return allChanges;
}
```
